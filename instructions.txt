# 1. Parse → Describe (skips cached) → Embed → Cluster into 5
python process_synthetic_batch.py \
    --recordings-dir sessions \
    --clusters 5 \
    --concurrency 5 \
    --skip-training-data
 
# 2. Build per-cluster training data
python build_training_data.py
 
# 3. Fine-tune 5 LoRA adapters (one per cluster)
python finetune_job.py --all-clusters --no-wandb --skip-inference
 
# =============================================
# 4. ON VM: Start GPU inference service (stays running permanently)
# =============================================
python service.py
# Listens on 0.0.0.0:8000
# Models stay loaded in GPU — no reload per request
# Swagger docs at http://<VM_IP>:8000/docs

# =============================================
# 5. ON LOCAL MACHINE: Run browser evaluation client
# =============================================
# This runs AgentQL + Playwright locally, calls VM for predictions.
# Each cluster model gets its own browser session.
# Closed feedback loop: error → retry with recovery action up to 3 times.

python local_client.py \
    --vm-url http://<VM_IP>:8000 \
    --target-url https://fun-city-xi.vercel.app \
    --description "Sign up for an account and comment on the top thread" \
    --app-name "FunCity" \
    --max-steps 20 \
    --max-retries 3

# Run specific clusters only:
python local_client.py \
    --vm-url http://<VM_IP>:8000 \
    --target-url https://fun-city-xi.vercel.app \
    --description "Browse posts and upvote content" \
    --clusters 0 2 4

# =============================================

curl http://<VM_IP>:8000/health
