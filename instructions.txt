# 1. Parse → Describe (skips cached) → Embed → Cluster into 5
python process_synthetic_batch.py \
    --recordings-dir sessions \
    --clusters 5 \
    --concurrency 5 \
    --skip-training-data
 
# 2. Build per-cluster training data
python build_training_data.py
 
# 3. Fine-tune 5 LoRA adapters (one per cluster)
python finetune_job.py --all-clusters --no-wandb --skip-inference
 
# =============================================
# 4. ON VM: Start GPU inference service with ngrok tunnel
# =============================================
# Install ngrok support
pip install pyngrok

# Get a free ngrok auth token from https://dashboard.ngrok.com/get-started/your-authtoken
# Start service with tunnel (public HTTPS URL, accessible from anywhere):
NGROK_AUTH_TOKEN=<your-ngrok-token> SERVICE_API_KEY=<your-secret> python service.py

# Output will show:
#   NGROK TUNNEL ACTIVE
#   Public URL: https://xxxx-xxxx.ngrok-free.app

pip install agentql playwright requests
playwright install chromium

python local_client.py \
